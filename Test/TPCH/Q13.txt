select sum(l_extendedprice*(1-l_discount)/1000) as revenue from ids as id1, ids as id2, supplier, lineitem, orders, customer, nation where supplier.S_SUPPKEY=lineitem.L_SUPPKEY and lineitem.L_ORDERKEY=orders.O_ORDERKEY and orders.O_CUSTKEY=customer.C_CUSTKEY and customer.C_NATIONKEY=nation.N_NATIONKEY and nation.N_NATIONKEY=supplier.S_NATIONKEY and id1.i_id = supplier.s_id and id2.i_id = customer.c_id  group by n_name;
