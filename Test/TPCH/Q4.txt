select n_name, extract(year FROM l_shipdate), sum(l_extendedprice*(1-l_discount)/1000) as revenue from supplier, lineitem, orders, customer, nation, ids as id1, ids as id2 where s_suppkey = l_suppkey and o_orderkey = l_orderkey and c_custkey = o_custkey and c_nationkey = n_nationkey and id1.i_id = supplier.s_id and id2.i_id = customer.c_id and ((l_shipdate between date '1995-01-01' and date '1995-02-01') or (l_shipdate between date '1996-01-01' and date '1996-02-08')) group by n_name, extract(year FROM l_shipdate);
